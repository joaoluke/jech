// ================================================
// 29 - Multiline REPL Test
// Demonstrates multi-line support: functions, conditions
// Demonstra suporte a múltiplas linhas: funções e condicionais
// ================================================

// --- Simple function with one parameter ---
// Função simples com um parâmetro
do greet(name) {
    say(name);
}

// --- Function with multiple parameters ---
// Função com múltiplos parâmetros
do showInfo(name, age, city) {
    say(name);
    say(age);
    say(city);
}

// --- Test the functions ---
// Testa as funções
greet("Alice");
showInfo("Bob", "30", "New York");

// --- Multi-line conditional ---
// Condicional de múltiplas linhas
keep x = 15;

when (x > 10) {
    say("X is greater than 10");
}
